; x86-64 Linux Assembly - Hello World
; This would be the assembly that generates machine code

section .data
    msg db 'Hello World', 0x0a
    len equ $ - msg

section .text
    global _start

_start:
    ; write(1, msg, len)
    mov rax, 1
    mov rdi, 1
    mov rsi, msg
    mov rdx, len
    syscall

    ; exit(0)
    mov rax, 60
    xor rdi, rdi
    syscall
